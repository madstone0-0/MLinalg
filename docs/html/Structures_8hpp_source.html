990 Matrix<number, m - 1, n - 1> subset(std::optional<size_t> i, std::optional<size_t> j) const {
991 static_assert(m == n, "Not a square matrix");
992 static_assert(m > 1 && n > 1, "Cannot subset");
993
994 Matrix<number, m - 1, n - 1> res;
995 size_t resRow = 0;
996 for (size_t k = 0; k < m; ++k) {
997 if (i.has_value() && i.value() == k) continue; // Skip the row to be removed
998
999 size_t resCol = 0;
1000 for (size_t z = 0; z < n; ++z) {
1001 if (j.has_value() && j.value() == z) continue; // Skip the column to be removed
1002
1003 res.at(resRow).at(resCol) = this->at(k).at(z);
1004 ++resCol;
1005 }
1006 ++resRow;
1007 }
1008 return res;
1009 }
