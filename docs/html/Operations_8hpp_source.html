100 template <Number number, size_t n>
101 optional<number> solveEquation(Row<number, n> row, size_t varPos, Row<optional<number>, n - 1> solutions) {
102 vector<number> leftSide{row.begin(), row.end() - 1};
103 vector<number> rightSide{row.back()};
104 for (size_t i{}; i < leftSide.size(); i++) {
105 optional<number> var = solutions.at(i);
106 if (i != varPos) {
107 number res = var.has_value() ? var.value() * leftSide.at(i) * -1 : leftSide.at(i) * -1;
108 rightSide.push_back(res);
109 }
110 }
111 leftSide = vector{leftSide.at(varPos)};
112 auto rightSimpl = std::accumulate(rightSide.begin(), rightSide.end(), number{0});
113 if (leftSide.at(0) == 0) {
114 if (rightSimpl == 0 && leftSide.at(0) == 0)
115 return 0;
116 else
117 return std::nullopt;
118 }
119
120 number sol = rightSimpl / leftSide.at(0);
121 return sol;
122 }

169 template <Number number, size_t m, size_t n>
170 LinearSystem<number, m, n> rearrangeSystem(LinearSystem<number, m, n> system) {
171 auto getZeroCount = [](const Row<number, n>& row) {
172 size_t mid = floor((n - 1) / 2.);
173 size_t count{};
174 for (size_t i{}; i < n - 1; i++)
175 if (row.at(i) == 0 && i <= mid) count++;
176 return count;
177 };
178
179 auto cmp = [](const std::pair<size_t, size_t>& a, const std::pair<size_t, size_t>& b) {
180 return (a.second < b.second);
181 };
182
183 auto getRowSortedCounts = [&getZeroCount, &cmp](const LinearSystem<number, m, n> sys) {
184 std::multimap<size_t, size_t> rowCounts{};
185 vector<std::pair<size_t, size_t>> res{};
186 for (size_t i{}; i < m; i++) rowCounts.emplace(i, getZeroCount(sys.at(i)));
187 auto it = rowCounts.begin();
188 for (; it != rowCounts.end(); it++) res.emplace_back(it->first, it->second);
189 std::sort(res.begin(), res.end(), cmp);
190
191 return res;
192 };
193
194 LinearSystem<number, m, n> rearranged{system};
195 auto rowCounts = getRowSortedCounts(system);
196 for (size_t i{}; i < rowCounts.size(); i++) {
197 const auto& [idx, count] = rowCounts.at(i);
198 if (i != idx) {
199 std::swap(rearranged.at(i), rearranged.at(idx));
200 }
201 rowCounts = getRowSortedCounts(rearranged);
202 }
203 return rearranged;
204 }

212 template <Number number, size_t m, size_t n>
213 LinearSystem<number, m, n> refSq(LinearSystem<number, m, n> system) {
214 RowOptional<number, n> pivots = getPivots(system);
215 while (!isInEchelonForm(system, pivots)) {
216 for (size_t i{}; i < pivots.size(); i++) {
217 pivots = getPivots(system);
218 const auto& pivot = pivots.at(i);
219 if (!pivot.has_value()) continue;
220 for (size_t j{i + 1}; j < m; j++) {
221 system = rearrangeSystem(system);
222 pivots = getPivots(system);
223 auto lower = system.at(j).at(i);
224 if (lower == 0) continue;
225 if (pivot == 0) continue;
226
227 auto permute = lower / pivot.value();
228 auto rowItems = system.at(j);
229 Row<number, n> permuted{permute * system.at(i)};
230 auto newRow = permuted - rowItems;
231 system.at(j) = newRow;
232 }
233 }
234 }
235 return system;
236 }

244 template <Number number, size_t m, size_t n>
245 LinearSystem<number, m, n> refRec(LinearSystem<number, m, n> system) {
246 for (size_t col = 0; col < n && col < m; ++col) {
247 system = rearrangeSystem(system);
248 size_t pivotRow = col;
249
250 for (size_t row = pivotRow + 1; row < m; ++row) {
251 if (system.at(row).at(col) != 0) {
252 auto factor = system.at(row).at(col) / system.at(pivotRow).at(col);
253 for (size_t j = col; j < n; ++j) {
254 system.at(row).at(j) -= factor * system.at(pivotRow).at(j);
255 }
256 }
257 }
258 }
259 return system;
260 }

284 template <Number number, size_t m, size_t n>
285 LinearSystem<number, m, n> rrefRec(LinearSystem<number, m, n> system, bool identity = true) {
286 RowOptional<number, n> pivots = getPivots(system);
287 if (!isInEchelonForm(system, pivots)) system = ref(system);
288
289 for (size_t col = n; col > 0; --col) {
290 system = rearrangeSystem(system);
291 size_t pivotRow = col - 1;
292
293 if (pivotRow >= m) continue;
294
295 if (system.at(pivotRow).at(col - 1) == 0) continue;
296
297 if (identity) {
298 auto pivotValue = system.at(pivotRow).at(col - 1);
299 if (pivotValue != 0) {
300 for (size_t j = 0; j < n; ++j) {
301 system.at(pivotRow).at(j) /= pivotValue;
302 }
303 }
304 }
305
306 for (size_t row = 0; row < pivotRow; ++row) {
307 auto upperValue = system.at(row).at(col - 1);
308 if (upperValue != 0) {
309 for (size_t j = 0; j < n; ++j) {
310 system.at(row).at(j) -= upperValue * system.at(pivotRow).at(j);
311 }
312 }
313 }
314 }
315
316 return system;
317 }

326 template <Number number, size_t m, size_t n>
327 LinearSystem<number, m, n> rrefSq(LinearSystem<number, m, n> system, bool identity = true) {
328 RowOptional<number, n> pivots = getPivots(system);
329 if (!isInEchelonForm(system, pivots)) system = ref(system);
330 pivots = getPivots(system);
331
332 while (!isInReducedEchelonForm(system, pivots)) {
333 auto size = pivots.size() < n - 1 ? pivots.size() : n - 1;
334 for (int i{1}; i < size; i++) {
335 pivots = getPivots(system);
336 const auto& pivot = pivots.at(i);
337 if (!pivot.has_value()) continue;
338 for (int j{i - 1}; j >= 0; j--) {
339 pivots = getPivots(system);
340 auto upper = system.at(j).at(i);
341 if (upper == 0) continue;
342 if (pivot == 0) continue;
343
344 auto permute = upper / pivot.value();
345 auto rowItems = system.at(j);
346 Row<number, n> permuted{permute * system.at(i)};
347 auto newRow = permuted - rowItems;
348 system.at(j) = newRow;
349 }
350 }
351 }
352
353 if (identity)
354 for (size_t i{}; i < pivots.size(); i++) {
355 try {
356 const auto& pivot{system.at(i).at(i)};
357 if (pivot == 0) continue;
358 if (pivot != 1) system.at(i) = system.at(i) * (1 / pivot);
359 } catch (const std::out_of_range& e) {
360 continue;
361 }
362 }
363
364 return system;
365 }

407 template <Number number, size_t m, size_t n>
408 optional<RowOptional<number, n>> findSolutions(LinearSystem<number, m, n> system) {
409 RowOptional<number, n> solutions{};
410 system = rearrangeSystem(system);
411
412 auto reducedEchelon = rref(system);
413
414 if (isInconsistent(reducedEchelon)) return nullopt;
415
416 for (auto i{solutions.rbegin()}; i != solutions.rend(); i++) {
417 try {
418 auto index = std::distance(solutions.rbegin(), i);
419 solutions.at(index) = solveEquation(reducedEchelon.at(index), index, solutions);
420 } catch (const std::out_of_range& e) {
421 continue;
422 }
423 }
424 return solutions;
425 }

447 template <Number number, size_t m, size_t n>
448 optional<Matrix<number, m, n>> inverse(const LinearSystem<number, m, n>& system) {
449 auto det = system.det();
450 if (det == 0) return std::nullopt;
451 if (m == 2 && n == 2)
452 return (1 / det) * Matrix<number, m, n>{{system.at(1).at(1), -system.at(0).at(1)},
453 {-system.at(1).at(0), system.at(0).at(0)}};
454 else {
455 auto identity = I<number, m>();
456 auto augmented = system.augment(identity);
457 auto rrefAug = rref(augmented);
458 auto inv = rrefAug.colToVectorSet();
459 inv.erase(inv.begin(), inv.begin() + m);
460 return mlinalg::structures::helpers::fromColVectorSet<number, m, n>(inv);
461 }
